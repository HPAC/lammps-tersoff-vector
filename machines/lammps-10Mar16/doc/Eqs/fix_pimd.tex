$$
  Z = \int d{\bf q} d{\bf p} \cdot \textrm{exp} [ -\beta H_{eff} ]
$$

$$
  H_{eff} = \bigg(\sum_{i=1}^P \frac{p_i^2}{2m_i}\bigg) + V_{eff}
$$

$$
  V_{eff} = \sum_{i=1}^P \bigg[ \frac{mP}{2\beta^2 \hbar^2} (q_i - q_{i+1})^2 + \frac{1}{P} V(q_i)\bigg]
$$
